create temp table sbr_rfm as

select accountnumber, rfm_recency*100 + rfm_frequency*10 + rfm_monetary as rfm_combined,

        case when rfm_recency*100 + rfm_frequency*10 + rfm_monetary between '111' and '144' then 'lost'

             when rfm_recency*100 + rfm_frequency*10 + rfm_monetary between '211' and '244' then 'at risk'

     when rfm_recency*100 + rfm_frequency*10 + rfm_monetary between '311' and '344' then 'slipping away'

     when rfm_recency*100 + rfm_frequency*10 + rfm_monetary between '411' and '444' then 'active'

end rfm_index,

case when rfm_recency*100 + rfm_frequency*10 + rfm_monetary between '111' and '144' then 'Revive interest with reach out campaign, ignore otherwise.'

             when rfm_recency*100 + rfm_frequency*10 + rfm_monetary between '211' and '244' then 'Send personalized emails to reconnect, offer renewals, provide helpful resources.'

     when rfm_recency*100 + rfm_frequency*10 + rfm_monetary between '311' and '344' then 'Make limited time offers, recommend based on past purchases. Reactivate them.'

     when rfm_recency*100 + rfm_frequency*10 + rfm_monetary between '411' and '444' then 'Reward them. Can be early adopters for new products. Will promote your brand.'

end rfm_actionable_tip

from (select accountnumber,

        ntile(4) over (order by redemptiondate) as rfm_recency,

        ntile(4) over (order by count_order) as rfm_frequency,

        ntile(4) over (order by pointsrequired) as rfm_monetary

from

        (select accountnumber, max(redemptiondate) as redemptiondate, count(*) as count_order, sum(pointsrequired) as pointsrequired from redemptions group by 1) as derivedTable) as derivedTable2;
